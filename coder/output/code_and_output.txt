```python
# Python code to calculate the first 10,000 terms of the series
total_sum = 0
for i in range(10000):
    term = 1 / (2 * i + 1)  # Calculate the odd number (1, 3, 5, ..., 19999)
    if i % 2 == 0:
        total_sum += term  # Add for even index
    else:
        total_sum -= term  # Subtract for odd index

# Multiply the total by 4 as required
total_sum *= 4

# Print the final result
print(total_sum)
```
